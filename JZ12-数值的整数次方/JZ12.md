## JZ12 数值的整数次方
### 描述：
给定一个double类型的浮点数base和int类型的整数exponent。求base的exponent次方。

保证base和exponent不同时为0。不得使用库函数，同时不需要考虑大数问题，也不用考虑小数点后面0的位数。

### 思路：
方法1：直接分情况处理
```C++
class Solution {
public:
    double Power(double base, int exponent) {
        double res = 1;
        if(exponent > 0){
            for(int i = 0; i < exponent; i++){
                res = res * base;
            }
        }
        else if(exponent < 0){
            exponent = 0 - exponent;
            for (int i = 0; i < exponent; i++){
                res = res * base;
            }
            res = 1 / res;
        }
        else{
            res = 1;
        }
        return res;
    }
};
```
方法2：递归
```C++
class Solution {
public:
    double Power(double base, int exponent) {
        if(exponent == 0){
            return 0;
        }
        if(exponent == 1){
            return base;
        }
        if(base < 0.00001 && base > -0.00001 && exponent < 0){
            return 0;
        }
        double res = Power(base, exponent >> 1);
        res *= res;
        if(exponent & 0x01 == 1){
            res *= base;
        }
        return res;
    }
};
```
